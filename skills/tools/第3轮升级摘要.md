### 解决方案2：问题生成器

```python
class ProductManagerQuestionGenerator:
    """产品经理问题生成器 - 永不停止质询"""
    
    QUESTION_CATEGORIES = [
        "代码质量",
        "安全性",
        "性能",
        "可维护性",
        "团队协作",
        "自动化",
        "智能化",
        "预测能力"
    ]
    
    def generate_next_round(self, completed_rounds):
        """生成下一轮问题"""
        questions = []
        
        # 从不同维度生成问题
        for category in self.QUESTION_CATEGORIES:
            q = self._generate_question_for_category(
                category, 
                difficulty_level=completed_rounds + 1
            )
            questions.append(q)
        
        return questions
```
